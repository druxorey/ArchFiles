$ sudo pacman -S vim neovim git neofetch reflector 

$ sudo cp /etc/pacman.d/mirrorlist /etc/pacman.d/mirrorlist.bak 

$ sudo reflector --verbose --latest 10 --protocol https --sort rate --save /etc/pacman.d/mirrorlist

$ sudo pacman -Sy

In 'sudo nano /etc/pacman.conf' Uncomment 'Color; ParallelDownloads = 5; IloveCandy'

$ sudo pacman -Sy
